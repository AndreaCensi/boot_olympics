    assert isinstance(spec, StreamSpec)

    spec2 = spec.from_yaml(y)

[docs]def check_pickling(x):
    spec = StreamSpec.from_yaml(x)
    sio = StringIO()
    pickle.dump(spec, sio)
    sio = StringIO(sio.getvalue())
    spec2 = pickle.load(sio)
    assert spec == spec2
